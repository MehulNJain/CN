//Server_Side
#include<stdio.h>
#include<sys/types.h>
#include<sys/socket.h>
#include<netinet/in.h>

int main(void)
{
    int sockfd, a, b, c, length, choice;
    struct sockaddr_in client, server;
    sockfd = socket(AF_INET, SOCK_DGRAM, 0);
    server.sin_family = AF_INET;
    server.sin_port = 6210;
    server.sin_addr.s_addr = inet_addr("127.0.0.1");
    bind(sockfd, (struct sockaddr *)&server, sizeof(server));
    length = sizeof(server);

    while (1) {
        recvfrom(sockfd, &a, sizeof(a), 0, (struct sockaddr*)&client, &length);
        recvfrom(sockfd, &b, sizeof(b), 0, (struct sockaddr*)&client, &length);
        recvfrom(sockfd, &choice, sizeof(choice), 0, (struct sockaddr*)&client, &length);

        switch (choice) {
            case 1: c = a + b;
                break;
            case 2: c = a - b;
                break;
            case 3: c = a * b;
                break;
            case 4: c = a / b;
                break;
        }

        sendto(sockfd, &c, sizeof(c), 0, (struct sockaddr*)&client, length);
        printf("\nResult: %d\t", c);
    }
}
